#include <stdio.h>

int main()
{
    int n, i, time = 0, completed = 0;
    int at[10], bt[10], rt[10], pr[10];
    int wt[10], tat[10], hp, min;

    printf("Enter number of processes: ");
    scanf("%d", &n);

    for(i = 0; i < n; i++)
    {
        printf("P%d (AT BT Priority): ", i + 1);
        scanf("%d%d%d", &at[i], &bt[i], &pr[i]);
        rt[i] = bt[i];
    }

    while(completed < n)
    {
        hp = -1;
        min = 999;

        for(i = 0; i < n; i++)
        {
            if(at[i] <= time && rt[i] > 0 && pr[i] < min)
            {
                min = pr[i];
                hp = i;
            }
        }

        if(hp == -1)
        {
            time++;
            continue;
        }

        rt[hp]--;
        time++;

        if(rt[hp] == 0)
        {
            completed++;
            tat[hp] = time - at[hp];
            wt[hp] = tat[hp] - bt[hp];
        }
    }

    printf("\nP\tWT\tTAT\n");

    for(i = 0; i < n; i++)
        printf("P%d\t%d\t%d\n", i + 1, wt[i], tat[i]);

    return 0;
}
