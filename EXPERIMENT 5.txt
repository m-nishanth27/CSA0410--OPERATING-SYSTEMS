#include <stdio.h>

int main()
{
    int n,i,j,temp;
    int bt[10],p[10],wt[10],tat[10];

    printf("Enter number of processes: ");
    scanf("%d",&n);

    printf("Enter Burst Time and Priority:\n");

    for(i=0;i<n;i++)
        scanf("%d%d",&bt[i],&p[i]);

    for(i=0;i<n-1;i++)
        for(j=i+1;j<n;j++)
            if(p[i]>p[j])
            {
                temp=p[i];
                p[i]=p[j];
                p[j]=temp;

                temp=bt[i];
                bt[i]=bt[j];
                bt[j]=temp;
            }

    wt[0]=0;

    for(i=1;i<n;i++)
        wt[i]=wt[i-1]+bt[i-1];

    printf("\nBT\tPriority\tWT\tTAT\n");

    for(i=0;i<n;i++)
    {
        tat[i]=wt[i]+bt[i];
        printf("%d\t%d\t\t%d\t%d\n",bt[i],p[i],wt[i],tat[i]);
    }

    return 0;
}
