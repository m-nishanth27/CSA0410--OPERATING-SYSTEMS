#include <stdio.h>

int main()
{
    int bt[10],rem[10];
    int n,tq,i,time=0,flag;
    int wt=0,tat=0;

    printf("Enter number of processes: ");
    scanf("%d",&n);

    printf("Enter Burst Time:\n");

    for(i=0;i<n;i++)
    {
        scanf("%d",&bt[i]);
        rem[i]=bt[i];
    }

    printf("Enter Time Quantum: ");
    scanf("%d",&tq);

    do
    {
        flag=0;

        for(i=0;i<n;i++)
        {
            if(rem[i]>0)
            {
                flag=1;

                if(rem[i]>tq)
                {
                    time+=tq;
                    rem[i]-=tq;
                }
                else
                {
                    time+=rem[i];
                    wt+=time-bt[i];
                    tat+=time;
                    rem[i]=0;
                }
            }
        }

    }while(flag);

    printf("Average Waiting Time=%.2f\n",(float)wt/n);
    printf("Average Turnaround Time=%.2f\n",(float)tat/n);

    return 0;
}
